1class Solution {
2public:
3    bool containsNearbyDuplicate(vector<int>& nums, int k) {
4        unordered_map<int,int>map;
5        // map store the nums values as keys and their indexes as a value of that key. in this problem we just do that we check the current value of nums is exist in the map or not if exist then we check that  current value address and the address of the key appear in the map their difference is less then or equal to k or not...
6        for(int i=0 ; i<nums.size(); i++){
7            if(map.count(nums[i])){
8                if(i-map[nums[i]]<=k){
9                    return true;
10                }
11            }
12            map[nums[i]]=i;
13        }
14        return false;
15    }
16};
